openssl req -new \
    -newkey rsa:2048 \
    -keyout kafka-6-creds/kafka-6.key \
    -out kafka-6-creds/kafka-6.csr \
    -config kafka-6-creds/kafka-6.cnf \
    -nodes

openssl x509 -req \
    -days 3650 \
    -in kafka-6-creds/kafka-6.csr \
    -CA ca.crt \
    -CAkey ca.key \
    -CAcreateserial \
    -out kafka-6-creds/kafka-6.crt \
    -extfile kafka-6-creds/kafka-6.cnf \
    -extensions v3_req

openssl pkcs12 -export \
    -in kafka-6-creds/kafka-6.crt \
    -inkey kafka-6-creds/kafka-6.key \
    -chain \
    -CAfile ca.pem \
    -name kafka-6 \
    -out kafka-6-creds/kafka-6.p12 \
    -password pass:your-password6

keytool -importkeystore \
    -deststorepass your-password \
    -destkeystore kafka-6-creds/kafka.kafka-6.keystore.pkcs12 \
    -srckeystore kafka-6-creds/kafka-6.p12 \
    -deststoretype PKCS12  \
    -srcstoretype PKCS12 \
    -noprompt \
    -srcstorepass your-password6


keytool -import \
    -file ca.crt \
    -alias ca \
    -keystore kafka-6-creds/kafka.kafka-6.truststore.jks \
    -storepass your-password6 \
     -noprompt
